## Overview

Once a proposal is ready, you can share it with clients through:

- **Email**: Send personalized emails with proposal link
- **PDF Export**: Download as PDF for offline sharing
- **Direct Link**: Copy shareable link for manual distribution
- **Client Access**: Clients view proposals without login

### Send Email API

```typescript
// apps/kitasuro/src/app/api/proposal/send-email/route.ts
export const POST = withAxiom(async (request: AxiomRequest) => {
  const session = await auth.api.getSession({ headers: await headers() });
  const orgId = await getOrganizationId(session);
  
  const body = await request.json();
  const { proposalId, recipientEmail, recipientName, subject, message, isTest } = body;

  // Fetch proposal
  const proposal = await db.query.proposals.findFirst({
    where: and(
      eq(proposals.id, proposalId),
      eq(proposals.organizationId, orgId)
    ),
    with: { organization: true, days: true },
  });

  // Generate proposal URL
  const proposalUrl = `${env.NEXT_PUBLIC_APP_URL}/proposal/${proposalId}`;
  
  // Render email HTML
  const html = renderProposalEmail({
    clientName: recipientName,
    agencyName: proposal.organization?.name,
    proposalTitle: proposal.tourTitle || proposal.name,
    proposalUrl,
    startDate,
    duration,
    message,
    isTest,
  });

  // Send via Resend
  const result = await resend.emails.send({
    from: env.RESEND_FROM_EMAIL,
    to: recipientEmail,
    subject: subject || `Your Travel Proposal: ${proposal.tourTitle}`,
    html,
  });

  return NextResponse.json({ success: true });
});
```

## PDF Export

### PDF Generation API

```typescript
// apps/kitasuro/src/app/api/proposal/[id]/pdf/route.ts
export const GET = withAxiom(async (
  request: AxiomRequest,
  { params }: { params: Promise<{ id: string }> }
) => {
  const { id } = await params;
  
  // Check authentication
  const session = await auth.api.getSession({ headers: await headers() });
  const orgId = await getOrganizationId(session);
  
  // Check plan access for PDF export
  const access = await checkFeatureAccess(orgId, 'pdfExport');
  if (!access.allowed) {
    return NextResponse.json({ error: access.reason }, { status: 403 });
  }
  
  // Fetch proposal with all related data
  const proposal = await db.query.proposals.findFirst({
    where: and(eq(proposals.id, id), eq(proposals.organizationId, orgId)),
    with: {
      client: true,
      organization: true,
      days: {
        with: {
          activities: true,
          accommodations: { with: { accommodation: true } },
          meals: true,
          nationalPark: true,
        },
        orderBy: (days, { asc }) => [asc(days.dayNumber)],
      },
    },
  });
  
  // Transform data for PDF
  const pdfData = {
    tourTitle: proposal.tourTitle || proposal.name,
    clientName: proposal.client?.name || 'Valued Client',
    agencyName: proposal.organization?.name || 'Travel Agency',
    startDate: proposal.startDate ? new Date(proposal.startDate) : undefined,
    duration: proposal.days.length,
    days: proposal.days.map((day, index) => ({
      dayNumber: day.dayNumber,
      date: startDate ? format(addDays(startDate, index), 'MMMM d, yyyy') : undefined,
      title: day.title,
      description: day.description,
      destination: day.nationalPark?.name,
      activities: day.activities.map((act) => ({
        name: act.name,
        moment: act.moment,
      })),
      accommodation: day.accommodations?.[0]?.accommodation?.name,
      meals: day.meals ? {
        breakfast: day.meals.breakfast,
        lunch: day.meals.lunch,
        dinner: day.meals.dinner,
      } : undefined,
    })),
    pricingRows: proposal.pricingRows,
    inclusions: proposal.inclusions,
    exclusions: proposal.exclusions,
  };
  
  // Generate PDF using @react-pdf/renderer
  const pdfBuffer = await renderToBuffer(
    ProposalPDF({ proposal: pdfData })
  );
  
  // Return PDF
  return new NextResponse(new Uint8Array(pdfBuffer), {
    headers: {
      'Content-Type': 'application/pdf',
      'Content-Disposition': `attachment; filename="${proposal.tourTitle}-${id}.pdf"`,
    },
  });
});
```

### Plan Access Control

PDF export is controlled by subscription plan:

```typescript
// lib/plans.ts
export async function checkFeatureAccess(
  orgId: string,
  feature: 'pdfExport' | 'comments' | 'activeProposals'
) {
  const plan = await getOrgPlan(orgId);
  
  if (feature === 'pdfExport') {
    if (!plan.limits.pdfExport) {
      return { 
        allowed: false, 
        reason: 'PDF export requires a paid plan' 
      };
    }
  }
  
  return { allowed: true };
}
```

## Client Access

### Public Viewing

Clients access proposals without login at `/proposal/[id]`:

```typescript
// apps/kitasuro/src/app/proposal/[id]/page.tsx
export default async function ItineraryPage({ params }: Props) {
  const { id } = await params;
  
  // Public endpoint - no authentication required
  const proposal = await trpc.proposals.getById({ id });
  
  if (!proposal) {
    notFound();
  }
  
  const transformedData = await transformProposalToItineraryData(proposal);
  const orgPlan = await getOrgPlan(proposal.organizationId);
  
  const commentsEnabled = orgPlan.limits.comments;
  const showWatermark = !orgPlan.limits.noWatermark;
  
  return (
    <CommentsProvider proposalId={id} readOnly={!commentsEnabled}>
      <CommentsOverlay>
        {transformedData.theme === 'kudu' ? (
          <KuduTheme data={transformedData} />
        ) : transformedData.theme === 'discovery' ? (
          <DiscoveryTheme data={transformedData} />
        ) : (
          <MinimalisticTheme data={transformedData} />
        )}
        
        {showWatermark && (
          <div className="fixed right-4 bottom-4 rounded-full bg-black/70 px-4 py-2 text-xs text-white">
            Powered by Kitasuro
          </div>
        )}
      </CommentsOverlay>
    </CommentsProvider>
  );
}
```

### Metadata Generation

```typescript
export async function generateMetadata({ params }: Props): Promise<Metadata> {
  const { id } = await params;
  const proposal = await getCachedProposal(id);
  
  const dayCount = proposal.days.length;
  const title = proposal.tourTitle || proposal.name;
  const orgName = proposal.organization?.name;
  
  const descriptionParts: string[] = [];
  if (dayCount > 0) descriptionParts.push(`${dayCount}-day`);
  if (proposal.tourType) descriptionParts.push(proposal.tourType.toLowerCase());
  if (proposal.countries?.length) descriptionParts.push(`across ${proposal.countries.join(', ')}`);
  if (proposal.startCity && proposal.endCity) {
    descriptionParts.push(`${proposal.startCity} → ${proposal.endCity}`);
  }
  
  return {
    title: orgName ? `${title} — ${orgName}` : title,
    description: descriptionParts.join(' '),
    robots: { index: false, follow: false },
    openGraph: {
      title: orgName ? `${title} — ${orgName}` : title,
      description: descriptionParts.join(' '),
      type: 'website',
    },
  };
}
```

## Best Practices

### Email Communication

- Always send test emails first
- Personalize the message for each client
- Include clear call-to-action (View Proposal button)
- Mention how to leave feedback (comments or reply)

### Link Sharing

- Use direct links for WhatsApp/SMS sharing
- Copy link for embedding in other platforms
- Share link only after proposal is finalized

### PDF Export

- Download PDF for offline presentations
- Send PDF as email attachment for clients without internet
- Keep PDF version for records

### Client Experience

- Ensure proposal is published before sharing
- Test proposal link before sending to client
- Enable comments for client feedback
- Monitor proposal views and engagement
