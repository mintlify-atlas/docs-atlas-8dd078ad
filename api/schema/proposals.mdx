## proposals

Main proposal table containing tour proposals for clients.

<ParamField path="id" type="text" required>
  Unique identifier for the proposal
</ParamField>

<ParamField path="name" type="text" required>
  Name of the proposal
</ParamField>

<ParamField path="tourId" type="uuid" required>
  Reference to the tour ID (cascade delete)
</ParamField>

<ParamField path="clientId" type="uuid">
  Reference to the client ID (set null on delete)
</ParamField>

<ParamField path="tourTitle" type="text">
  Title of the tour
</ParamField>

<ParamField path="tourType" type="text">
  Type of tour (e.g., safari, adventure, cultural)
</ParamField>

<ParamField path="theme" type="text">
  Visual theme for the proposal. Default: `minimalistic`
</ParamField>

<ParamField path="heroImage" type="text">
  URL to the hero image for the proposal
</ParamField>

<ParamField path="startDate" type="timestamp">
  Start date of the tour
</ParamField>

<ParamField path="startCity" type="text">
  Starting city for the tour
</ParamField>

<ParamField path="startCityLat" type="numeric">
  Latitude of the starting city (precision: 10, scale: 7)
</ParamField>

<ParamField path="startCityLng" type="numeric">
  Longitude of the starting city (precision: 10, scale: 7)
</ParamField>

<ParamField path="endCity" type="text">
  Ending city for the tour
</ParamField>

<ParamField path="endCityLat" type="numeric">
  Latitude of the ending city (precision: 10, scale: 7)
</ParamField>

<ParamField path="endCityLng" type="numeric">
  Longitude of the ending city (precision: 10, scale: 7)
</ParamField>

<ParamField path="pickupPoint" type="text">
  Pickup location for the tour
</ParamField>

<ParamField path="transferIncluded" type="text">
  Transfer inclusion details
</ParamField>

<ParamField path="pricingRows" type="json">
  Pricing breakdown as JSON array
  
  ```typescript
  Array<{ id: string; count: number; type: string; unitPrice: number }>
  ```
</ParamField>

<ParamField path="extras" type="json">
  Additional extras and add-ons as JSON array
  
  ```typescript
  Array<{ id: string; name: string; price: number; selected: boolean }>
  ```
</ParamField>

<ParamField path="travelerGroups" type="json">
  Traveler group composition as JSON array
  
  ```typescript
  Array<{ id: string; count: number; type: string }>
  ```
</ParamField>

<ParamField path="countries" type="text[]">
  Array of countries covered in the tour
</ParamField>

<ParamField path="inclusions" type="text[]">
  Array of items included in the tour
</ParamField>

<ParamField path="exclusions" type="text[]">
  Array of items not included in the tour
</ParamField>

<ParamField path="hidePricing" type="boolean">
  Whether to hide pricing in the proposal. Default: `false`
</ParamField>

<ParamField path="organizationId" type="uuid">
  Reference to the organization ID
</ParamField>

<ParamField path="status" type="proposal_status" required>
  Status of the proposal. Default: `draft`
  
  **Enum values**: `draft`, `shared`
</ParamField>

<ParamField path="createdAt" type="timestamp" required>
  Timestamp when the proposal was created (auto-generated)
</ParamField>

<ParamField path="updatedAt" type="timestamp" required>
  Timestamp when the proposal was last updated (auto-generated)
</ParamField>

### Relations

- **tour**: Many-to-one relationship with `tours` table
- **organization**: Many-to-one relationship with `organizations` table
- **client**: Many-to-one relationship with `clients` table
- **days**: One-to-many relationship with `proposalDays` table
- **comments**: One-to-many relationship with `comments` table
- **notes**: One-to-many relationship with `proposalNotes` table
- **assignments**: One-to-many relationship with `proposalAssignments` table
